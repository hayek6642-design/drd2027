## 1. Global State Manager - App Modes

**`src/state/appMode.js`**
```javascript
/**
 * App Mode Manager
 * Global state management for application modes
 * idle | onboarding | chat | listening
 */

import { useState, useCallback, useRef } from 'react';
import { logger } from '../utils/logger';

// Mode definitions with their allowed transitions
const MODE_TRANSITIONS = {
  idle: ['onboarding', 'chat', 'listening'],
  onboarding: ['chat', 'idle'],
  chat: ['listening', 'idle', 'onboarding'],
  listening: ['chat', 'idle', 'onboarding']
};

class AppModeManager {
  constructor() {
    this.currentMode = 'idle';
    this.previousMode = null;
    this.listeners = new Set();
    this.modeData = {}; // Store extra data for each mode
  }

  subscribe(listener) {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }

  notify(mode, previousMode, data = {}) {
    this.listeners.forEach(listener => {
      try {
        listener(mode, previousMode, data);
      } catch (err) {
        logger.error('Mode listener error:', err);
      }
    });
  }

  canTransition(toMode) {
    const allowed = MODE_TRANSITIONS[this.currentMode] || [];
    return allowed.includes(toMode);
  }

  transition(toMode, data = {}) {
    if (!this.canTransition(toMode)) {
      logger.warn(`Invalid transition: ${this.currentMode} → ${toMode}`);
      return false;
    }

    this.previousMode = this.currentMode;
    this.currentMode = toMode;
    this.modeData[toMode] = data;

    logger.log(`Mode transition: ${this.previousMode} → ${toMode}`, data);
    this.notify(toMode, this.previousMode, data);
    return true;
  }

  getMode() {
    return this.currentMode;
  }

  getPreviousMode() {
    return this.previousMode;
  }

  getModeData(mode = null) {
    const target = mode || this.currentMode;
    return this.modeData[target] || {};
  }

  is(mode) {
    return this.currentMode === mode;
  }

  isNot(mode) {
    return this.currentMode !== mode;
  }

  // Quick transition methods
  toIdle(data) {
    return this.transition('idle', data);
  }

  toOnboarding(data) {
    return this.transition('onboarding', data);
  }

  toChat(data) {
    return this.transition('chat', data);
  }

  toListening(data) {
    return this.transition('listening', data);
  }

  // Reset
  reset() {
    this.currentMode = 'idle';
    this.previousMode = null;
    this.modeData = {};
    this.notify('idle', null, {});
  }
}

// Singleton instance
const appMode = new AppModeManager();

// React Hook
export function useAppMode() {
  const [mode, setMode] = useState(appMode.getMode());
  const [modeData, setModeData] = useState(appMode.getModeData());

  const transition = useCallback((toMode, data) => {
    return appMode.transition(toMode, data);
  }, []);

  const toIdle = useCallback((data) => appMode.toIdle(data), []);
  const toOnboarding = useCallback((data) => appMode.toOnboarding(data), []);
  const toChat = useCallback((data) => appMode.toChat(data), []);
  const toListening = useCallback((data) => appMode.toListening(data), []);

  useState(() => {
    const unsubscribe = appMode.subscribe((newMode, prevMode, data) => {
      setMode(newMode);
      setModeData(data);
    });
    return unsubscribe;
  }, []);

  return {
    mode,
    modeData,
    previousMode: appMode.getPreviousMode(),
    is: appMode.is.bind(appMode),
    isNot: appMode.isNot.bind(appMode),
    transition,
    toIdle,
    toOnboarding,
    toChat,
    toListening,
    canTransition: appMode.canTransition.bind(appMode)
  };
}

export default appMode;
```

**`src/onboarding/onboardingEngine.js`**
```javascript
/**
 * Onboarding Engine
 * Conversational, voice-driven onboarding flow
 * Human-like interaction with sequential questions
 */

import { logger } from '../utils/logger';
import memoryManager from '../memory/memoryManager';
import appMode from '../state/appMode';
import voiceOutput from '../voice/voiceOutput';

// Onboarding flow configuration
const ONBOARDING_FLOW = {
  intro: {
    id: 'intro',
    zagelMessage: "Hi... I am Zagel 🕊\nCan I get to know you?",
    expectYesNo: true,
    onYes: 'name',
    onNo: 'exit',
    saveAnswer: false
  },
  name: {
    id: 'name',
    zagelMessage: "What is your name?",
    field: 'name',
    saveAnswer: true,
    confirmation: (value) => `Nice to meet you, ${value}!`,
    next: 'job'
  },
  job: {
    id: 'job',
    zagelMessage: "What do you do?",
    field: 'job',
    saveAnswer: true,
    confirmation: (value) => `Interesting! ${value} sounds like a great field.`,
    next: 'interests'
  },
  interests: {
    id: 'interests',
    zagelMessage: "What are your interests?",
    field: 'interests',
    saveAnswer: true,
    isArray: true, // Can have multiple interests
    confirmation: (value) => `Great! We'll explore ${Array.isArray(value) ? value.join(', ') : value} together.`,
    next: 'complete'
  },
  complete: {
    id: 'complete',
    zagelMessage: (userData) => `Perfect! I'm ready to help you, ${userData.name}. Just tap the circle anytime you need me.`,
    saveAnswer: false,
    isFinal: true
  }
};

class OnboardingEngine {
  constructor() {
    this.currentStep = null;
    this.userData = {
      name: '',
      job: '',
      interests: []
    };
    this.isActive = false;
    this.onStepComplete = null;
    this.onOnboardingComplete = null;
    this.onOnboardingExit = null;
  }

  /**
   * Start the onboarding flow
   */
  async start(options = {}) {
    logger.log('Starting onboarding...');

    this.onStepComplete = options.onStepComplete;
    this.onOnboardingComplete = options.onOnboardingComplete;
    this.onOnboardingExit = options.onOnboardingExit;

    // Check if user already has data
    const existing = await memoryManager.getUserPreference('name');
    if (existing && !options.force) {
      logger.log('User already onboarded, skipping');
      if (options.onOnboardingComplete) {
        options.onOnboardingComplete(await this.getStoredUserData());
      }
      return false;
    }

    this.isActive = true;
    this.userData = { name: '', job: '', interests: [] };
    
    // Start with intro
    await this.goToStep('intro');
    return true;
  }

  /**
   * Navigate to a specific step
   */
  async goToStep(stepId) {
    const step = ONBOARDING_FLOW[stepId];
    if (!step) {
      logger.error('Invalid onboarding step:', stepId);
      return;
    }

    this.currentStep = step;

    // Generate message (can be dynamic)
    const message = typeof step.zagelMessage === 'function' 
      ? step.zagelMessage(this.userData)
      : step.zagelMessage;

    logger.log('Onboarding step:', stepId, message);

    // Notify UI
    if (this.onStepComplete) {
      this.onStepComplete({
        step: stepId,
        zagelMessage: message,
        expectYesNo: step.expectYesNo || false,
        isFinal: step.isFinal || false,
        userData: { ...this.userData }
      });
    }

    // Speak the message
    await voiceOutput.speak(message);

    // If not yes/no and not final, auto-start listening
    if (!step.expectYesNo && !step.isFinal) {
      // Signal to start listening after speech
      setTimeout(() => {
        appMode.toListening({ 
          context: 'onboarding',
          step: stepId,
          autoStart: true 
        });
      }, 500);
    }

    return step;
  }

  /**
   * Handle user answer
   */
  async handleUserAnswer(answer) {
    if (!this.isActive || !this.currentStep) {
      logger.warn('No active onboarding step');
      return;
    }

    const step = this.currentStep;
    logger.log('Handling answer for step:', step.id, answer);

    // Handle yes/no for intro
    if (step.expectYesNo) {
      const normalized = this.normalizeYesNo(answer);
      
      if (normalized === 'yes') {
        await this.goToStep(step.onYes);
      } else {
        await this.exitOnboarding();
      }
      return;
    }

    // Validate and save answer
    if (step.saveAnswer && step.field) {
      const validated = this.validateAnswer(answer, step);
      
      if (!validated.isValid) {
        // Re-ask with clarification
        await voiceOutput.speak("I didn't catch that. " + step.zagelMessage);
        appMode.toListening({ context: 'onboarding', step: step.id });
        return;
      }

      // Save to local data
      if (step.isArray) {
        // Parse comma-separated or single value
        const values = answer.split(/[،,]/).map(s => s.trim()).filter(Boolean);
        this.userData[step.field] = values;
      } else {
        this.userData[step.field] = validated.value;
      }

      // Save to persistent memory immediately
      await this.saveToMemory(step.field, this.userData[step.field]);

      // Speak confirmation
      const confirmation = typeof step.confirmation === 'function'
        ? step.confirmation(this.userData[step.field])
        : step.confirmation;
      
      if (confirmation) {
        await voiceOutput.speak(confirmation);
      }
    }

    // Move to next step
    if (step.isFinal) {
      await this.completeOnboarding();
    } else if (step.next) {
      // Small delay for natural conversation flow
      setTimeout(() => {
        this.goToStep(step.next);
      }, 800);
    }
  }

  /**
   * Validate user answer
   */
  validateAnswer(answer, step) {
    const trimmed = answer.trim();
    
    if (!trimmed) {
      return { isValid: false, value: null };
    }

    // Name validation
    if (step.field === 'name') {
      // Remove extra spaces, keep only letters and spaces
      const cleaned = trimmed.replace(/[^\p{L}\s]/gu, '').trim();
      if (cleaned.length < 2) {
        return { isValid: false, value: null };
      }
      return { isValid: true, value: cleaned };
    }

    // Job validation
    if (step.field === 'job') {
      if (trimmed.length < 3) {
        return { isValid: false, value: null };
      }
      return { isValid: true, value: trimmed };
    }

    // Interests validation
    if (step.field === 'interests') {
      const values = trimmed.split(/[،,]/).map(s => s.trim()).filter(Boolean);
      if (values.length === 0) {
        return { isValid: false, value: null };
      }
      return { isValid: true, value: values };
    }

    return { isValid: true, value: trimmed };
  }

  /**
   * Normalize yes/no answers
   */
  normalizeYesNo(answer) {
    const normalized = answer.trim().toLowerCase();
    
    const yesWords = ['yes', 'yeah', 'yep', 'sure', 'ok', 'okay', 'أيوة', 'نعم', 'تمام', 'ماشي', 'أه', 'yes please', 'why not'];
    const noWords = ['no', 'nope', 'nah', 'لا', 'مش دلوقتي', 'not now', 'maybe later', 'بعدين'];
    
    if (yesWords.some(w => normalized.includes(w))) return 'yes';
    if (noWords.some(w => normalized.includes(w))) return 'no';
    
    // Default to yes if unclear but positive sounding
    return normalized.length > 3 ? 'yes' : 'no';
  }

  /**
   * Save to persistent memory
   */
  async saveToMemory(field, value) {
    try {
      // Update user preferences
      const update = { [field]: value };
      await memoryManager.updatePreferences(update);
      
      // Also save as user fact for richer context
      await memoryManager.addToConversation('system', `User ${field}: ${Array.isArray(value) ? value.join(', ') : value}`, {
        type: 'onboarding_fact',
        field: field
      });
      
      logger.log('Saved to memory:', field, value);
    } catch (err) {
      logger.error('Failed to save to memory:', err);
    }
  }

  /**
   * Complete onboarding
   */
  async completeOnboarding() {
    logger.log('Onboarding complete!', this.userData);
    this.isActive = false;

    // Final save
    await memoryManager.updatePreferences({
      onboardingComplete: true,
      onboardingCompletedAt: Date.now(),
      ...this.userData
    });

    // Transition to chat mode
    appMode.toChat({ 
      source: 'onboarding_complete',
      userData: { ...this.userData }
    });

    if (this.onOnboardingComplete) {
      this.onOnboardingComplete({ ...this.userData });
    }
  }

  /**
   * Exit onboarding early
   */
  async exitOnboarding() {
    logger.log('Onboarding exited by user');
    this.isActive = false;
    
    await voiceOutput.speak("No problem! You can always tell me about yourself later. Just tap the circle when you need me.");
    
    appMode.toIdle({ source: 'onboarding_exit' });

    if (this.onOnboardingExit) {
      this.onOnboardingExit();
    }
  }

  /**
   * Get stored user data
   */
  async getStoredUserData() {
    const prefs = await memoryManager.getMemory();
    return {
      name: prefs.userPreferences.name || '',
      job: prefs.userPreferences.job || '',
      interests: prefs.userPreferences.interests || [],
      onboardingComplete: prefs.userPreferences.onboardingComplete || false
    };
  }

  /**
   * Check if onboarding is needed
   */
  async isOnboardingNeeded() {
    const userData = await this.getStoredUserData();
    return !userData.onboardingComplete || !userData.name;
  }

  /**
   * Replay onboarding (for settings)
   */
  async replay(options = {}) {
    return this.start({ ...options, force: true });
  }

  /**
   * Skip onboarding
   */
  async skip() {
    logger.log('Skipping onboarding');
    await memoryManager.updatePreferences({ onboardingSkipped: true });
    this.exitOnboarding();
  }

  /**
   * Get current state
   */
  getState() {
    return {
      isActive: this.isActive,
      currentStep: this.currentStep?.id || null,
      userData: { ...this.userData }
    };
  }
}

// Singleton
const onboardingEngine = new OnboardingEngine();

export default onboardingEngine;
```

```javascript
/**
 * Voice Output
 * Text-to-Speech for Zagel responses
 */

import * as Speech from 'expo-speech';
import { logger } from '../utils/logger';

class VoiceOutput {
  constructor() {
    this.isSpeaking = false;
    this.speechQueue = [];
    this.currentOptions = {};
    
    // Default TTS options
    this.defaultOptions = {
      language: 'ar-SA', // Arabic Saudi (good for Egyptian/Levantine too)
      pitch: 1.0,
      rate: 0.9, // Slightly slower for clarity
      volume: 1.0,
      onStart: () => {
        this.isSpeaking = true;
        logger.log('TTS started');
      },
      onDone: () => {
        this.isSpeaking = false;
        this.processQueue();
        logger.log('TTS done');
      },
      onError: (err) => {
        this.isSpeaking = false;
        logger.error('TTS error:', err);
        this.processQueue();
      }
    };
  }

  /**
   * Speak text
   */
  async speak(text, options = {}) {
    if (!text) return;

    // Queue if already speaking
    if (this.isSpeaking) {
      this.speechQueue.push({ text, options });
      logger.log('Queued TTS:', text.substring(0, 30));
      return;
    }

    // Merge options
    const speakOptions = {
      ...this.defaultOptions,
      ...options,
      onStart: () => {
        this.isSpeaking = true;
        options.onStart?.();
        this.defaultOptions.onStart();
      },
      onDone: () => {
        this.isSpeaking = false;
        options.onDone?.();
        this.defaultOptions.onDone();
      },
      onError: (err) => {
        this.isSpeaking = false;
        options.onError?.(err);
        this.defaultOptions.onError(err);
      }
    };

    try {
      // Detect language (simple heuristic)
      const isArabic = /[\u0600-\u06FF]/.test(text);
      speakOptions.language = isArabic ? 'ar-SA' : (options.language || 'en-US');

      // Adjust tone based on content
      if (options.tone) {
        speakOptions = this.applyTone(speakOptions, options.tone);
      }

      logger.log('Speaking:', text.substring(0, 50));
      Speech.speak(text, speakOptions);
      
    } catch (err) {
      logger.error('Failed to speak:', err);
      this.isSpeaking = false;
    }
  }

  /**
   * Process queued speech
   */
  processQueue() {
    if (this.speechQueue.length > 0 && !this.isSpeaking) {
      const next = this.speechQueue.shift();
      setTimeout(() => this.speak(next.text, next.options), 100);
    }
  }

  /**
   * Stop speaking
   */
  async stop() {
    try {
      await Speech.stop();
      this.isSpeaking = false;
      this.speechQueue = [];
    } catch (err) {
      logger.error('Failed to stop TTS:', err);
    }
  }

  /**
   * Apply tone modifications
   */
  applyTone(options, tone) {
    const toneSettings = {
      friendly: { pitch: 1.1, rate: 0.95 },
      serious: { pitch: 0.9, rate: 0.85 },
      excited: { pitch: 1.2, rate: 1.0 },
      calm: { pitch: 1.0, rate: 0.8 },
      funny: { pitch: 1.15, rate: 1.0 }
    };

    const settings = toneSettings[tone] || toneSettings.friendly;
    return { ...options, ...settings };
  }

  /**
   * Check if speaking
   */
  isCurrentlySpeaking() {
    return this.isSpeaking;
  }

  /**
   * Get available voices
   */
  async getVoices() {
    try {
      const voices = await Speech.getAvailableVoicesAsync();
      return voices;
    } catch (err) {
      logger.error('Failed to get voices:', err);
      return [];
    }
  }
}

// Singleton
const voiceOutput = new VoiceOutput();

export default voiceOutput;
```

```javascript
/**
 * Toast Message Component
 * Shows temporary notifications with animations
 */

import React, { useEffect, useRef } from 'react';
import { View, Text, StyleSheet, Animated } from 'react-native';

export default function ToastMessage({ message, visible, onHide, duration = 3000 }) {
  const fadeAnim = useRef(new Animated.Value(0)).current;
  const slideAnim = useRef(new Animated.Value(-100)).current;

  useEffect(() => {
    if (visible) {
      // Show animation
      Animated.parallel([
        Animated.timing(fadeAnim, {
          toValue: 1,
          duration: 300,
          useNativeDriver: true,
        }),
        Animated.spring(slideAnim, {
          toValue: 0,
          friction: 8,
          useNativeDriver: true,
        }),
      ]).start();

      // Auto hide
      const timer = setTimeout(() => {
        hide();
      }, duration);

      return () => clearTimeout(timer);
    }
  }, [visible]);

  const hide = () => {
    Animated.parallel([
      Animated.timing(fadeAnim, {
        toValue: 0,
        duration: 200,
        useNativeDriver: true,
      }),
      Animated.timing(slideAnim, {
        toValue: -100,
        duration: 200,
        useNativeDriver: true,
      }),
    ]).start(() => {
      onHide?.();
    });
  };

  if (!visible && fadeAnim.__getValue() === 0) return null;

  return (
    <Animated.View 
      style={[
        styles.container,
        {
          opacity: fadeAnim,
          transform: [{ translateY: slideAnim }]
        }
      ]}
    >
      <View style={styles.content}>
        <Text style={styles.icon}>🕊</Text>
        <Text style={styles.text}>{message}</Text>
      </View>
    </Animated.View>
  );
}
```

**`src/ui/ZagelButton.js`**
```javascript
/**
 * Zagel Circle Button
 * Animated floating trigger with pulse/glow effects
 */

import React, { useEffect, useRef, useCallback } from 'react';
import { 
  TouchableOpacity, 
  View, 
  StyleSheet, 
  Animated,
  Vibration,
  Dimensions 
} from 'react-native';

const { width } = Dimensions.get('window');

export default function ZagelButton({ 
  onPress, 
  isActive = false,
  isListening = false,
  showGlow = true 
}) {
  const scaleAnim = useRef(new Animated.Value(1)).current;
  const glowAnim = useRef(new Animated.Value(0)).current;
  const pulseAnim = useRef(new Animated.Value(1)).current;
  const rotateAnim = useRef(new Animated.Value(0)).current;

  // Breathing animation (idle)
  useEffect(() => {
    if (!isActive && !isListening) {
      const breathe = Animated.loop(
        Animated.sequence([
          Animated.timing(scaleAnim, {
            toValue: 1.05,
            duration: 2000,
            useNativeDriver: true,
          }),
          Animated.timing(scaleAnim, {
            toValue: 1,
            duration: 2000,
            useNativeDriver: true,
          }),
        ])
      );
      breathe.start();
      return () => breathe.stop();
    }
  }, [isActive, isListening]);

  // Glow animation
  useEffect(() => {
    if (showGlow) {
      const glow = Animated.loop(
        Animated.sequence([
          Animated.timing(glowAnim, {
            toValue: 1,
            duration: 1500,
            useNativeDriver: true,
          }),
          Animated.timing(glowAnim, {
            toValue: 0,
            duration: 1500,
            useNativeDriver: true,
          }),
        ])
      );
      glow.start();
      return () => glow.stop();
    }
  }, [showGlow]);

  // Active pulse animation
  useEffect(() => {
    if (isActive || isListening) {
      const pulse = Animated.loop(
        Animated.sequence([
          Animated.timing(pulseAnim, {
            toValue: 1.3,
            duration: 800,
            useNativeDriver: true,
          }),
          Animated.timing(pulseAnim, {
            toValue: 1,
            duration: 800,
            useNativeDriver: true,
          }),
        ])
      );
      pulse.start();
      return () => pulse.stop();
    } else {
      pulseAnim.setValue(1);
    }
  }, [isActive, isListening]);

  // Rotation for active state
  useEffect(() => {
    if (isActive) {
      const rotate = Animated.loop(
        Animated.timing(rotateAnim, {
          toValue: 1,
          duration: 3000,
          useNativeDriver: true,
        })
      );
      rotate.start();
      return () => rotate.stop();
    }
  }, [isActive]);

  const handlePress = useCallback(() => {
    // Haptic feedback
    Vibration.vibrate(50);
    
    // Scale animation on press
    Animated.sequence([
      Animated.timing(scaleAnim, {
        toValue: 0.9,
        duration: 100,
        useNativeDriver: true,
      }),
      Animated.timing(scaleAnim, {
        toValue: 1,
        duration: 100,
        useNativeDriver: true,
      }),
    ]).start();
```

```javascript
  const { classify, extract } = useInstructionClassifier();

  // Initialize
  useEffect(() => {
    memoryManager.init();
    
    // Subscribe to mode changes
    const unsubscribe = appMode.subscribe((newMode, prevMode, data) => {
      handleModeChange(newMode, prevMode, data);
    });
    
    return unsubscribe;
  }, []);

  // Handle voice transcript changes
  useEffect(() => {
    if (transcript && !isListening && mode === 'listening') {
      handleVoiceInput(transcript);
    }

  useEffect(() => {
    scrollViewRef.current?.scrollToEnd({ animated: true });

    logger.log('Mode changed:', prevMode, '→', newMode);
    
    switch (newMode) {
      case 'onboarding':
        setOnboardingVisible(true);
        break;
      case 'chat':
        setOnboardingVisible(false);
        break;
      case 'listening':

    // Check if we need onboarding
    const needsOnboarding = await onboardingEngine.isOnboardingNeeded();
    
    if (needsOnboarding) {
      // Start onboarding
      startOnboarding();
    } else {
      // Go to chat mode
      toChat({ source: 'zagel_button' });
      
      // Greet user by name
      const userData = await onboardingEngine.getStoredUserData();
      const greeting = `Let's work on something interesting today, ${userData.name}!`;
      
      addMessage('assistant', greeting);
      voiceOutput.speak(greeting);
    }

  const startOnboarding = () => {

        setOnboardingStep(stepData.step);
        setOnboardingMessage(stepData.zagelMessage);

      await onboardingEngine.handleUserAnswer(text);

      await handleChatInput(text);
    }

  const handleChatInput = async (text) => {
    if (!text.trim()) return;

    // Add user message
    addMessage('user', text);

    // Classify
    const classification = classify(text);
    
    if (classification.isInstruction) {
      // Handle instruction
      await handleInstruction(text, classification);
    } else {
      // Send to Zagel
      await handleZagelMessage(text);

      }
    } else {

      await startListening('ar-SA');
    }
  }, [isListening, stopListening, startListening, mode]);

            <TextInput

              onSubmitEditing={handleTextSubmit}
              returnKeyType="send"
            />
            <TouchableOpacity onPress={handleTextSubmit} style={styles.sendButton}>

            </TouchableOpacity>
            <TouchableOpacity
```

```javascript
import React from 'react';
import { View, Text, StyleSheet, Animated } from 'react-native';

export default function StatusIndicator({ status }) {
  const pulseAnim = React.useRef(new Animated.Value(1)).current;
  
  React.useEffect(() => {
    if (status === 'listening') {
      Animated.loop(
        Animated.sequence([
          Animated.timing(pulseAnim, {
            toValue: 1.5,
            duration: 800,
            useNativeDriver: true,
          }),
          Animated.timing(pulseAnim, {
            toValue: 1,
            duration: 800,
            useNativeDriver: true,
          }),
        ])
      ).start();
    } else {
      pulseAnim.setValue(1);
    }
  }, [status]);

  const getStatusConfig = () => {
    switch (status) {
      case 'idle':
        return { text: 'انقر الدائرة', color: '#64748b', icon: '◉' };
      case 'onboarding':
        return { text: 'التعريف', color: '#f59e0b', icon: '👋' };
      case 'listening':
        return { text: 'جاري الاستماع...', color: '#ef4444', icon: '●' };
      case 'chat':
        return { text: 'جاهز', color: '#00d4ff', icon: '◉' };
      default:
        return { text: 'جاهز', color: '#00d4ff', icon: '◉' };
    }
  };

  const config = getStatusConfig();

  return (
    <View style={styles.container}>
      {status === 'listening' && (
        <Animated.View 
          style={[
            styles.pulse,
            { 
              backgroundColor: config.color,
              transform: [{ scale: pulseAnim }],
              opacity: pulseAnim.interpolate({
                inputRange: [1, 1.5],
                outputRange: [0.5, 0]
              })
            }
          ]} 
        />
      )}
      <View style={[styles.dot, { backgroundColor: config.color }]} />
      <Text style={[styles.text, { color: config.color }]}>
        {config.text}
      </Text>
    </View>
  );
}
```
